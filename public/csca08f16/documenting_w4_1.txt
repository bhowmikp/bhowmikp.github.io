def add_two_numbers(first, second):
    '''(float/int, float/int) -> float/int
    
    Returns the value of two numbers added together
    
    REQ-> None
    
    >>> add_two_numbers(1,2)
    3
    >>> add_two_numbers(1,3)
    4
    '''
    # calculate addition of the numbers
    add = first + second
    # retrun addition of two number
    return add
